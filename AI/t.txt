with open("./train/column_mapping.json", "r", encoding="utf-8") as f:
    COLUMN_MAP = json.load(f)


def map_text_to_column(user_text):
    user_text = user_text.lower()
    matched = {}

    for col_name, keywords in COLUMN_MAP.items():
        for kw in keywords:
            if kw in user_text:
                matched[col_name] = kw
                break

    return matched
